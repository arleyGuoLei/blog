    public static void Three() {
        // 第三题 数字魔方，输入3,5,7,9,11,13任意奇数，显示数字魔方
        System.out.println("请输入一个基数:");
        Scanner r = new Scanner(System.in);
        int n = r.nextInt();
        if (n % 2 == 0) {

        int[][] a = new int[n][n];
        int row = 0;
        int lin = n / 2;
        for (int i = 1; i <= n * n; i++) {
            a[row][lin] = i;
            row--;
            lin++;
            if (row == -1 && lin < n)
                row = n - 1;
            else if (row >= 0 && lin == n)
                lin = 0;
            else if ((row == -1 && lin == n) || (a[row][lin] != 0)) {
                row += 2;
                lin--;
            }
        }

    public static void SevenEight(boolean type) {
        // 排序  type = ture 选择排序   false 冒泡排序
        int[] arr = new int[10];
        int i = 0;
        Scanner r = new Scanner(System.in);
        while (i < 10) {
            if (r.hasNextInt()) {
                arr[i] = r.nextInt();
                i++;
            }
        }
        System.out.println("交换之前：");
        printIntArr(arr);
        if (type) {
            select(arr);
        } else {
            maoPao(arr);
        }

    }

    private static void select(int[] arr) {
        for (int i = 0; i < arr.length - 1; i++) {// 做第i趟排序
            int k = i;
            for (int j = k + 1; j < arr.length; j++) {// 选最小的记录
                if (arr[j] < arr[k]) {
                    k = j; //记下目前找到的最小值所在的位置
                }
            }
            //在内层循环结束，也就是找到本轮循环的最小的数以后，再进行交换
            if (i != k) {  //交换a[i]和a[k]
                int temp = arr[i];
                arr[i] = arr[k];
                arr[k] = temp;
            }
        }

        System.out.println("交换后：");
        printIntArr(arr);
    }

    private static void maoPao(int[] arr) {
        for (int i = 0; i < arr.length - 1; i++) {//外层循环控制排序趟数
            for (int j = 0; j < arr.length - 1 - i; j++) {//内层循环控制每一趟排序多少次
                if (arr[j] > arr[j + 1]) {
                    int temp = arr[j];
                    arr[j] = arr[j + 1];
                    arr[j + 1] = temp;
                }
            }
        }
        System.out.println("交换后：");
        printIntArr(arr);
    }

    private static void printIntArr(int[] arr) {
        for (int num : arr) {
            System.out.print(num + " ");
        }
    }
