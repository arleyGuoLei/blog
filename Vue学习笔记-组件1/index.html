计数器例子

1.Vue.component
2.this.$emit(‘change’)
3.@change=(“函数名”)

<!DOCTYPE html>
<html lang="en">
<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <meta http-equiv="X-UA-Compatible" content="ie=edge">
    <title>Counter</title>
    <script src="vue.js"></script>
</head>
<body>
    <div id="root">
        <item-counter ref="one" @change="handleChange"></item-counter>
        <item-counter ref="two" @change="handleChange"></item-counter>
        <div>{{sum}}</div>
    </div>
    <script>
        Vue.component('item-counter',{
            template:'<div @click="bindCounter">{{counter}}</div>',
            data:function(){
                return{
                    counter:0
                }
            },
            methods:{
                bindCounter:function(){
                    this.counter++;
                    this.$emit('change');
                }
            }
        })
        var vm = new Vue({
            el:'#root',
            data:{
                sum:0
            },
            methods:{
                handleChange:function(){

                    this.sum = this.$refs.one.counter + this.$refs.two.counter;
                }
            }

        })
    </script>
</body>
</html>

2 父子组件间的数据传递

  1. 局部组件

先定义一个对象,然后在new Vue里的components去引用

  1. 父组件向子组件传递值

:绑定 比如 :count=”xxx” 向模板传入count参数,需要在模板定义的时候在props进行定义count

  1. 单项数据流(子组件禁止直接去修改父组件的值)

父组件的数据可能被多个地方调用,所以禁止修改,需要用自己的变量去接收,然后在修改.

  1. 子元素给父元素传值(带参数)

函数绑定的地方直接写函数名,不去要参数,然后在用的时候直接存入,在函数定义时候接收.

计数器2
<!DOCTYPE html>
<html lang="en">

<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <meta http-equiv="X-UA-Compatible" content="ie=edge">
    <title>Counter</title>
    <script src="vue.js"></script>
</head>

<body>
    <div id="root">
        <item-counter :count="1" @inc="numberInc"></item-counter>
        <item-counter :count="1" @inc="numberInc"></item-counter>
        <div>{{total}}</div>
    </div>
    <script>
        var counter = {
            // @click="handCounter" 得写在模板上面,不能写在<item-counter xxx(这) :count="1"></item-counter>
            template: '<div @click="handCounter">{{number}}</div>',
            props: ['count'],
            data: function () {
                return{
                    // 这里得return   拷贝一份父元素的number
                    number: this.count
                }
            },
            methods:{
                handCounter:function(){
                    this.number++;
                    this.$emit('inc',1);
                }
            }

        }


        var vm = new Vue({
            el: '#root',
            components: { 'item-counter':counter },
            data:{
                total:2
            },
            methods:{
                numberInc:function(count){
                    this.total+=count;
                }
            }
        })
    </script>
</body>

</html>

5 非父子组件之间的传值问题

1.bus/发布订阅模式/观察者模式

this.bus.$emit()

<!DOCTYPE html>
<html lang="en">

<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <meta http-equiv="X-UA-Compatible" content="ie=edge">
    <title>非父子组件间的传值</title>
    <script src="vue.js"></script>
</head>

<body>
    <div id="root">
        <!-- 这里的content不能打: -->
        <item-child content="Arley"></item-child>
        <item-child content="GuoLei"></item-child>
    </div>
    <script>
       //关键↓ 1
        Vue.prototype.bus = new Vue();

        Vue.component('item-child', {
            props: {
                content: String
            },
            data: function () {
                return {
                    selfContent: this.content
                }
            },
            template: '<div @click="changeItem">{{selfContent}}</div>',
            methods:{
                changeItem:function(){
                   //关键↓ 2
                    this.bus.$emit('change',this.selfContent)
                }
            },
            //关键↓ 3
            mounted:function(){
                var that = this;
                // 4 监听
                this.bus.$on('change',function(msg){
                    that.selfContent = msg;
                })
            },
        })
        var vm = new Vue({
            el: '#root'
        })
    </script>
</body>

</html>

7 作用域插槽

子组件中留一部分代码,由父组件传入,决定渲染模式.

    <div id="root">
        <body-content>
            <template slot-scope="props">
                <li>{{props.item}} -- hello</li>
            </template>
        </body-content>
    </div>
    <script>
        Vue.component('body-content', {
            data: function () {
                return {
                    list: [1, 2, 3, 4, 5]
                }
            },
            template: `
            <div>
                <ul>
                    <slot v-for="item in list" :item=item></slot>
                </ul>
            </div>
            `
        })
        var vm = new Vue({
            el: '#root'
        })

    </script>

8 动态组件与 v-once 指令

<!DOCTYPE html>
<html lang="en">

<head>
    <meta charset="UTF-8">
    <meta name="viewport" content="width=device-width, initial-scale=1.0">
    <meta http-equiv="X-UA-Compatible" content="ie=edge">
    <title>插槽</title>
    <script src="vue.js"></script>
</head>

<body>
    <div id="root">
        <!-- 写法1 low -->
        <!-- <body-content-one v-if="type==='body-content-one'"></body-content-one>
        <body-content-two v-if="type==='body-content-two'"></body-content-two> -->

        <!-- 直接根据 is的属性赋值模板 -->
        <component :is="type"></component>
        <input type="button" value="切换" @click="change">
    </div>
    <script>
        // v-once 能把组件保存在内容,第一次渲染之后不用再次渲染
        Vue.component('body-content-one', {
            template: "<div v-once>one</div>"

        })
        Vue.component('body-content-two', {
            template: "<div v-once>two</div>"

        })
        var vm = new Vue({
            el: '#root',
            data:{
              type:'body-content-one'  
            },
            methods:{
                change:function(){
                    this.type = (this.type==='body-content-one'?'body-content-two':'body-content-one');
                }
            }
        })

    </script>
</body>

</html>
